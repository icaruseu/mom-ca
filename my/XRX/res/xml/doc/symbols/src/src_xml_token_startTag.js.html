 13 /**
 14  * Constructs a new start tag token.
 15  * @constructor
 16  * @extends xrx.token
 17  */
 18 xrx.token.StartTag = function(label, opt_offset, opt_length) {
 19   goog.base(this, xrx.token.START_TAG, label, opt_offset, opt_length);
 20 };
 21 goog.inherits(xrx.token.StartTag, xrx.token);

 25 /**
 26  * Compares the generic type of two tokens.
 27  *
 28  * @param {!number} type The type to check against.
 29  * @return {!boolean}
 30  */
 31 xrx.token.StartTag.prototype.typeOf = function(type) {
 32   return this.type_ === type || xrx.token.START_EMPTY_TAG === type || 
 33       xrx.token.TAG === type;
 34 };
