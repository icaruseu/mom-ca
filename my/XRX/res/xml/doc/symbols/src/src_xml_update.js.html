 15 xrx.update = function(stream, token, string) {
 16   var diff = string.length - token.length();
 17 
 18   stream.update(token.offset(), token.length(), string);
 19 
 20   token.length(string.length);
 21 
 22 
 23   
 24   return diff;
 25 };
